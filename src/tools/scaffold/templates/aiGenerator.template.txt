import { generateWithSchema } from '../../../services/geminiClient';
import { GeneratorOptions } from '../../../types';

/**
 * {{title}} AI Üretici
 */
export const generate{{enumKey}}FromAI = async (options: GeneratorOptions) => {
  const { difficulty, itemCount = 10 } = options;

  const prompt = `
    [ROL: {{logic.aiPrompt.role}}]
    GÖREV: {{logic.aiPrompt.task}}
    ZORLUK: ${difficulty}
    SAYI: ${itemCount}

    KURALLAR:
    {{#each logic.aiPrompt.rules}}
    - {{this}}
    {{/each}}
    
    ÇIKTI FORMATI (JSON):
    {{json logic.aiPrompt.schema}}
    `;

  const schema = {
    type: 'OBJECT',
    properties: {
      instruction: { type: 'STRING' },
      pedagogicalNote: { type: 'STRING' },
      items: {
        type: 'ARRAY',
        items: {{json logic.aiPrompt.schema.properties.items.items}}
      }
    },
    required: ['instruction', 'items', 'pedagogicalNote']
  };

  const result = await generateWithSchema(prompt, schema);
  return {
    type: '{{enumKey}}',
    title: '{{title}}',
    ...result
  };
};
